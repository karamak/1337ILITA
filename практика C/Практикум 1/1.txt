#include <locale.h>
#include <conio.h>
#include <stdio.h>

using namespace std;
int main()
{
	setlocale(LC_ALL, "Russian");

	char sex;
	float growth, weight;

	printf("Введите пол: (М/Ж)\n");
	scanf_s("%c", &sex);
	scanf_s("%*[^\n]");

	printf("Введите рост:\n");
	scanf_s("%f", &growth);
	scanf_s("%*[^\n]");
	
	printf("Введите вес:\n");
	scanf_s("%f", &weight);
	scanf_s("%*[^\n]");

	growth /= 100;
	const float IMT = weight / (growth*growth);
	printf("Ваш индекс массы тела: %f\n", IMT);

	if (IMT <18.5)
		printf("Нужно потолстеть");
	else if (IMT >= 18.5 && IMT < 25.0)
		printf("Вес в норме");
	else
		printf("Нужно похудеть");

	_getch();
	return 0;
}
