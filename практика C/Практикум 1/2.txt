#include <locale.h>
#include <conio.h>
#include <stdio.h>

using namespace std;
int main()
{
	setlocale(LC_ALL, "Russian");

	unsigned int hours, min, sec;
	while (true) {
		printf("Введите время: (ЧЧ:ММ:СС)\n");
		scanf_s("%2u%*c%2u%*c%2u", &hours, &min, &sec);
		scanf_s("%*[^\n]");

		if ((hours >= 0 && hours < 24) && (min >= 0 && min < 60) && (sec >= 0 && sec < 60)) break;

		printf("Неправильный ввод, введите ещё раз\n");
	}

	if (hours >= 6 && hours < 12) printf("Доброе утро!");
	else if (hours >= 12 && hours < 18) printf("Добрый день!");
	else if (hours >= 18 && hours < 22) printf("Добрый вечер!");
	else if (hours >= 22 && hours < 24 || hours >= 0 && hours < 6) printf("Доброй ночи!");


	_getch();
	return 0;
}
